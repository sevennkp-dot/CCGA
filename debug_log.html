2. ตรวจสอบคอลัมน์ในตาราง stock_movement_log

กำลังตรวจสอบ...

หากคอลัมน์ต่อไปนี้แสดงว่าไม่มี (Missing): operator, dispatched, dispatched_by, dispatched_at

คุณต้องคัดลอกคำสั่ง SQL ด้านล่างไปรันใน Supabase SQL Editor:

-- รันคำสั่งนี้เพื่อเพิ่มคอลัมน์ที่ขาดหายไปและสิทธิ์การใช้งาน ALTER TABLE public.stock_movement_log ADD COLUMN IF NOT EXISTS operator text, ADD COLUMN IF NOT EXISTS dispatched boolean DEFAULT false, ADD COLUMN IF NOT EXISTS dispatched_by text, ADD COLUMN IF NOT EXISTS dispatched_at timestamptz; -- สร้าง Index เพื่อความรวดเร็ว CREATE INDEX IF NOT EXISTS idx_stock_movement_log_dispatched ON stock_movement_log (dispatched, created_at DESC); -- ตรวจสอบและเปิดการใช้งาน RLS ALTER TABLE public.stock_movement_log ENABLE ROW LEVEL SECURITY; DROP POLICY IF EXISTS "Authenticated users can insert stock movement logs" ON public.stock_movement_log; CREATE POLICY "Authenticated users can insert stock movement logs" ON public.stock_movement_log FOR INSERT TO authenticated WITH CHECK (true); DROP POLICY IF EXISTS "Authenticated users can view stock movement logs" ON public.stock_movement_log; CREATE POLICY "Authenticated users can view stock movement logs" ON public.stock_movement_log FOR SELECT TO authenticated USING (true); DROP POLICY IF EXISTS "Authenticated users can update stock movement logs" ON public.stock_movement_log; CREATE POLICY "Authenticated users can update stock movement logs" ON public.stock_movement_log FOR UPDATE TO authenticated USING (true) WITH CHECK (true); -- โหลด Schema Cache ใหม่เพื่อให้ Supabase รู้จักคอลัมน์ใหม่ทันที NOTIFY pgrst, 'reload schema';
